-- Tabla de órdenes a procesar.
-- El estado se mueve con UPDATE condicional (patrón lease); nunca se bloquea con FOR UPDATE.

CREATE TABLE ordenes (
    id           BIGSERIAL    PRIMARY KEY,
    estado       VARCHAR(20)  NOT NULL DEFAULT 'PENDIENTE',  -- PENDIENTE/EN_PROCESO/COMPLETADA/FALLIDA
    reclamada_en TIMESTAMPTZ,                                -- cuándo se reclamó (base del lease)
    token        VARCHAR(64),                                -- dueño del lease actual
    contenido    TEXT,                                       -- datos de la orden (puede ser JSON; usa JSONB si prefieres)
    creada_en    TIMESTAMPTZ  NOT NULL DEFAULT now()
);

-- Índice plano (portable a H2/MySQL) que soporta bien la consulta de candidatas:
--   WHERE estado = 'PENDIENTE' OR (estado = 'EN_PROCESO' AND reclamada_en < :lease) ORDER BY creada_en
CREATE INDEX idx_ordenes_estado_creada ON ordenes (estado, creada_en);
